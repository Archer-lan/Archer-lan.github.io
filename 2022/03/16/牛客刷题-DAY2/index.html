class Solution{
public:
    ListNode* revers(ListNode* tail,ListNode* head){
        if(head==tail) return head;
        ListNode* pre=nullptr;
        ListNode* cur=head;
        while(cur!=tail){
			ListNode* next=cur->next;
            cur->next=pre;
            pre=cur;
            cur=next;
        }
        return pre;
    }
    ListNode* reverseGroup(ListNode* head,int k){
        if(!head) return head;
        ListNode* tail=head;
        for(int i=0;i<k;i++){
            if(!tail) return head;
            tail=tail->next;
        }
        ListNode* newHead=revers(head,tail);
        head->next=reverseGroup(tail,k);
        return newHead;
    }
}

class Solution {
public:
    ListNode* Merge(ListNode* pHead1, ListNode* pHead2) {
        ListNode* L=new ListNode(0);
        ListNode* s=L;
        while(pHead1!=nullptr&&pHead2!=nullptr){
            if(pHead1->val<pHead2->val){
                s->next=pHead1;
                s=s->next;
                pHead1=pHead1->next;
            }
            else{
                s->next=pHead2;
                s=s->next;
                pHead2=pHead2->next;
            }
        }
        if(pHead1==nullptr){
            s->next=pHead2;
        }
        else{
            s->next=pHead1;
        }
        return L->next;
    }
};
