class Solution {
public:
    bool Find(int target, vector<vector<int> > array) {
        int row=array.size();
        if(row==0||array[0].size()==0) return false;
        for(int i=0;i<row;i++){
            int right=array[0].size();
            int left=0;
            int mid=(right+left)/2;
            while(left<=right){
                if(array[i][mid]==target){
                    return true;
                }else if(array[i][mid]>target){
                    right=mid-1;
                    mid=(left+right)/2;
                }else{
                    left=mid+1;
                    mid=(left+right)/2;
                }
            }
        }
        return false;
    }
};

class Solution {
public:
    int mod = 1000000007;
    int mergeSort(int left, int right, vector<int>& data, vector<int>& temp){
        if (left >= right)    // åœæ­¢åˆ’åˆ†
            return 0;
        int mid = (left + right) / 2; //å–ä¸­é—´
        int res = mergeSort(left, mid, data, temp) + mergeSort(mid + 1, right, data, temp); //å·¦å³åˆ’åˆ†
        res %= mod;  //é˜²æ­¢æº¢å‡º
        int i = left, j = mid + 1;
        for (int k = left; k <= right; k++)
            temp[k] = data[k];
        for (int k = left; k <= right; k++) {
            if (i == mid + 1)
                data[k] = temp[j++];
            else if (j == right + 1 || temp[i] <= temp[j])
                data[k] = temp[i++];
            else { //å·¦è¾¹æ¯”å³è¾¹å¤§ï¼Œç­”æ¡ˆå¢åŠ 
                data[k] = temp[j++];
                res += mid - i + 1; // ç»Ÿè®¡é€†åºå¯¹
            }
        }
        return res % mod;
    }
    int InversePairs(vector<int> data) {
        int n = data.size();
        vector<int> res(n);
        return mergeSort(0, n - 1, data, res);
    }
};
