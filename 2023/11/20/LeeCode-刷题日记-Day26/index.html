/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
var reverseKGroup = function(head, k) {
    let pre=new ListNode(0);
    pre.next=head;
    let p=pre;
    while(head){
        let tail=p;
        for(let i=0;i<k;++i){
            tail=tail.next;
            if(!tail)
                return pre.next;
        }
        let next=tail.next;
        [head,tail]=Reverse(head,tail);
        p.next=head;
        tail.next=next;
        p=tail;
        head=tail.next;
    }
    return pre.next;
};

        }
        return [tail,head];

var reorderList = function(head) {
    if(head===null) return head;
    let p=head;
    let queue=[];
    while(p){
        queue.push(p);
        p=p.next;
    }
    while(queue.length>2){
        let pre=queue.shift();
        let tail=queue.pop();
        tail.next=pre.next;
        pre.next=tail;
    }
    queue[queue.length-1].next=null;
    return head;
};
