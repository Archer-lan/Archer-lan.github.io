/**
 * @param {string} s
 * @return {string}

var frequencySort = function(s) {
    let res=new Map();
    for(let i=0;i<s.length;i++){
        res.set(s[i],res.has(s[i])?res.get(s[i])+1:1);
    }
    let arr=Array.from(res).sort((a,b)=>{return b[1]-a[1]});
    let r='';
    for(let i of arr){
        for(let j=0;j<i[1];j++){
            r+=i[0];
        }
    }
    return r;
};

/**
 * @param {number[][]} points
 * @param {number} k
 * @return {number[][]}
 */
var kClosest = function(points, k) {
    let map=new Map()
    for(let num of points){
        let dis=Math.sqrt(Math.pow(num[0],2)+Math.pow(num[1],2));
        map.set(num,dis);
    }
    let arr=Array.from(map).sort((a,b)=>{return a[1]-b[1]});
    let res=[];
    for(let i=0;i<k;i++){
        res.push(arr[i][0])
    }
    return res;
};
